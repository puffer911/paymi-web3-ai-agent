docker run -d \
  --name n8n \
  --memory="256m" \
  --cpus="0.25" \
  -p 5678:5678 \
  -e N8N_BASIC_AUTH_ACTIVE=true \
  -e N8N_BASIC_AUTH_USER="" \
  -e N8N_BASIC_AUTH_PASSWORD="" \
  -v n8n_data:/home/node/.n8n \
  -e N8N_SECURE_COOKIE=false \
  -e N8N_RUNNERS_ENABLED=true \
  --restart unless-stopped \
  docker.n8n.io/n8nio/n8n
